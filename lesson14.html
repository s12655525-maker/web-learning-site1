Урок 14: DOM Манипуляции

⏱️ 50 минут 📊 Средний 📄 JavaScript

🔍 Поиск элементов

// По ID const el1 = document.getElementById("myId"); // По селектору (один) const el2 = document.querySelector(".class"); const el3 = document.querySelector("#id"); // По селектору (все) const els1 = document.querySelectorAll(".class"); const els2 = document.getElementsByTagName("div"); const els3 = document.getElementsByClassName("class");

📝 Изменение содержимого

const el = document.querySelector("div"); // Текст el.textContent = "Новый текст"; el.innerText = "Текст с учётом стилей"; // HTML el.innerHTML = "Жирный текст"; // Атрибуты el.id = "newId"; el.className = "newClass"; el.setAttribute("data-value", "123"); el.getAttribute("data-value"); // "123"

🎨 Работа со стилями

const el = document.querySelector("div"); // Прямой стиль el.style.color = "red"; el.style.backgroundColor = "blue"; el.style.fontSize = "20px"; // Классы el.classList.add("active"); el.classList.remove("hidden"); el.classList.toggle("visible"); el.classList.contains("active"); // true // Все классы el.className = "class1 class2";

➕ Создание и удаление

// Создание const div = document.createElement("div"); div.textContent = "Новый элемент"; div.className = "new-element"; // Добавление document.body.appendChild(div); parent.insertBefore(div, sibling); parent.append(div); // Множественное добавление // Удаление div.remove(); parent.removeChild(div); // Замена parent.replaceChild(newElement, oldElement);

👨‍👩‍👧 Навигация по DOM

const el = document.querySelector("div"); // Родитель и дети el.parentElement; el.children; // Только элементы el.childNodes; // Все узлы // Соседи el.previousElementSibling; el.nextElementSibling; el.previousSibling; el.nextSibling; // Первый/последний ребёнок el.firstElementChild; el.lastElementChild;
