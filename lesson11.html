Урок 11: Функции

⏱️ 40 минут 📊 Средний 📄 JavaScript

🔧 Объявление функций

// Function Declaration function greet(name) { return `Привет, ${name}!`; } // Function Expression const greet2 = function(name) { return `Привет, ${name}!`; }; // Arrow Function const greet3 = (name) => `Привет, ${name}!`; const greet4 = name => `Привет, ${name}!`; // Краткая форма

📥 Параметры и аргументы

function sum(a, b = 0) { // b - значение по умолчанию return a + b; } sum(5, 3); // 8 sum(5); // 5 // Остаточные параметры function sumAll(...numbers) { return numbers.reduce((acc, num) => acc + num, 0); } sumAll(1, 2, 3, 4); // 10

🔄 Область видимости

let globalVar = "глобальная"; function example() { let localVar = "локальная"; console.log(globalVar); // Доступна console.log(localVar); // Доступна } console.log(localVar); // Ошибка! Не доступна

⚡ Замыкания

function createCounter() { let count = 0; return function() { count++; return count; }; } const counter = createCounter(); console.log(counter()); // 1 console.log(counter()); // 2

🎯 Методы функций

// call function greet(greeting, punctuation) { console.log(`${greeting}, ${this.name}${punctuation}`); } const person = { name: "Иван" }; greet.call(person, "Привет", "!"); // Привет, Иван! // apply greet.apply(person, ["Здравствуйте", "."]); // Здравствуйте, Иван. // bind const boundGreet = greet.bind(person); boundGreet("Добрый день", "?"); // Добрый день, Иван?
