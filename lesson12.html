Урок 12: Массивы

⏱️ 45 минут 📊 Средний 📄 JavaScript

📋 Создание массивов

const arr1 = [1, 2, 3, 4, 5]; const arr2 = new Array(1, 2, 3); const arr3 = Array.of(1, 2, 3); const arr4 = Array(5).fill(0); // [0, 0, 0, 0, 0]

🔍 Основные методы

const fruits = ["яблоко", "банан", "груша"]; // Длина fruits.length; // 3 // Добавление/удаление fruits.push("апельсин"); // Добавить в конец fruits.pop(); // Удалить с конца fruits.unshift("киви"); // Добавить в начало fruits.shift(); // Удалить с начала // Поиск fruits.indexOf("банан"); // 1 fruits.includes("груша"); // true fruits.find(item => item.length > 5); // "яблоко"

🔄 Итерация

const numbers = [1, 2, 3, 4, 5]; // forEach numbers.forEach((num, index) => { console.log(`Индекс ${index}: ${num}`); }); // map - создать новый массив const doubled = numbers.map(num => num * 2); // [2, 4, 6, 8, 10] // filter - отфильтровать const evens = numbers.filter(num => num % 2 === 0); // [2, 4] // reduce - свёртка const sum = numbers.reduce((acc, num) => acc + num, 0); // 15

✂️ Методы изменения

const arr = [1, 2, 3, 4, 5]; // slice - вырезать (не изменяет оригинал) arr.slice(1, 3); // [2, 3] // splice - удалить/вставить (изменяет оригинал) arr.splice(1, 2); // Удалить 2 элемента с индекса 1 arr.splice(1, 0, 10); // Вставить 10 на позицию 1 // concat - объединение const arr2 = [6, 7]; const combined = arr.concat(arr2); // [1, 2, 3, 4, 5, 6, 7]

🔧 Сортировка

const nums = [3, 1, 4, 1, 5, 9]; nums.sort(); // [1, 1, 3, 4, 5, 9] (по строкам) nums.sort((a, b) => a - b); // [1, 1, 3, 4, 5, 9] (по числам) nums.reverse(); // [9, 5, 4, 3, 1, 1] const words = ["банан", "яблоко", "груша"]; words.sort(); // ["банан", "груша", "яблоко"]
