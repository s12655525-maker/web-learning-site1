Урок 16: Условия и циклы

⏱️ 35 минут 📊 Лёгкий 📄 JavaScript

🔀 Условные операторы

// if/else if (condition) { // код } else if (otherCondition) { // код } else { // код } // Тернарный оператор const result = condition ? "да" : "нет"; // Логическое И const value = obj && obj.prop; // Если obj существует // Логическое ИЛИ const name = inputName || "Гость"; // Значение по умолчанию // Опциональная цепочка const value = obj?.prop?.nested; // Безопасный доступ

🔄 Оператор switch

const day = "понедельник"; switch (day) { case "понедельник": console.log("Рабочий день"); break; case "субботка": case "воскресенье": console.log("Выходной"); break; default: console.log("Неизвестный день"); }

🔁 Цикл for

// Классический for for (let i = 0; i < 5; i++) { console.log(i); } // for...of (для массивов) const arr = [1, 2, 3]; for (const item of arr) { console.log(item); } // for...in (для объектов) const obj = { a: 1, b: 2 }; for (const key in obj) { console.log(key, obj[key]); }

⏮️ Циклы while

// while let i = 0; while (i < 5) { console.log(i); i++; } // do...while (выполнится хотя бы раз) let j = 0; do { console.log(j); j++; } while (j < 5);

⏹️ Управление циклами

for (let i = 0; i < 10; i++) { if (i === 3) { continue; // Пропустить итерацию } if (i === 7) { break; // Выйти из цикла } console.log(i); } // Вывод: 0, 1, 2, 4, 5, 6

🎯 Логические операторы

// И (&&) if (a > 0 && b > 0) { } // ИЛИ (||) if (a > 0 || b > 0) { } // НЕ (!) if (!isTrue) { } // Nullish coalescing const value = input ?? "default"; // Только для null/undefined
