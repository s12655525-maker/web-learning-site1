Урок 13: Объекты

⏱️ 40 минут 📊 Средний 📄 JavaScript

🏗️ Создание объектов

// Литерал объекта const person = { name: "Иван", age: 25, greet: function() { console.log(`Привет, ${this.name}!`); } }; // Конструктор const person2 = new Object(); person2.name = "Анна"; // Object.create const proto = { greet() { console.log("Привет!"); } }; const obj = Object.create(proto);

🔑 Доступ к свойствам

const user = { name: "Иван", "user age": 25 }; // Точечная нотация user.name; // "Иван" // Квадратные скобки user["name"]; // "Иван" user["user age"]; // 25 (только так для составных ключей) // Динамический доступ const key = "name"; user[key]; // "Иван"

🛠️ Методы объектов

const obj = { a: 1, b: 2, c: 3 }; // Ключи, значения, записи Object.keys(obj); // ["a", "b", "c"] Object.values(obj); // [1, 2, 3] Object.entries(obj); // [["a", 1], ["b", 2], ["c", 3]] // Копирование const copy = Object.assign({}, obj); const copy2 = { ...obj }; // Spread оператор // Слияние const merged = { ...obj, d: 4 };

🔒 Дескрипторы свойств

const obj = {}; Object.defineProperty(obj, "name", { value: "Иван", writable: true, // Можно изменять enumerable: true, // Видно в перечислении configurable: true // Можно удалить/изменить }); // Геттеры и сеттеры const user = { _name: "Иван", get name() { return this._name; }, set name(value) { this._name = value; } };

🔄 Методы объекта

const calculator = { result: 0, add(num) { this.result += num; return this; // Цепочка вызовов }, multiply(num) { this.result *= num; return this; } }; calculator.add(5).multiply(2).result; // 10
