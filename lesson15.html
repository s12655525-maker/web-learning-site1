🔧 Добавление обработчиков

const btn = document.querySelector("button"); // Метод addEventListener (рекомендуется) btn.addEventListener("click", function(event) { console.log("Клик!", event); }); // Стрелочная функция btn.addEventListener("click", (e) => { e.preventDefault(); // Предотвратить действие по умолчанию console.log("Клик!"); }); // Именованная функция function handleClick(e) { console.log("Клик!"); } btn.addEventListener("click", handleClick); // Удаление обработчика btn.removeEventListener("click", handleClick);

📋 Объект события

document.addEventListener("click", function(e) { console.log(e.type); // "click" console.log(e.target); // Элемент, на который кликнули console.log(e.currentTarget); // Элемент с обработчиком // Координаты console.log(e.clientX, e.clientY); // Относительно viewport console.log(e.pageX, e.pageY); // Относительно документа // Модификаторы console.log(e.ctrlKey); console.log(e.shiftKey); console.log(e.altKey); });

🔄 Делегирование событий

// Вместо обработчика на каждом элементе document.querySelectorAll(".item").forEach(item => { item.addEventListener("click", handleClick); }); // Один обработчик на родителе document.querySelector(".container").addEventListener("click", function(e) { if (e.target.classList.contains("item")) { handleClick(e); } });

⚡ Custom Events

// Создание события const event = new CustomEvent("myEvent", { detail: { message: "Привет!" } }); // Отправка события document.dispatchEvent(event); // Прослушивание события document.addEventListener("myEvent", function(e) { console.log(e.detail.message); // "Привет!" });
